create table genre (
	id SERIAL primary key,
	name varchar(60) not null
)
	
create table artist (
	id SERIAL primary key,
	name varchar(60) not null
)
	
create table GenreArtist (
	genre_Id int not null references genre(id) on delete cascade ,
	artist_Id int not null references artist(id) on delete cascade,
	primary key(genre_Id, artist_Id)
)


create table album (
	id SERIAL primary key,
	name varchar(120) not null,
	year_release date not null
)

create table albumArtist (
	artist_Id int not null references artist(id) on delete cascade,
	album_Id int not null references album(id) on delete cascade,
	primary key (album_Id, artist_Id)
)


create table track (
	id SERIAL primary key,
	name varchar(120) not null,
	duration int not null,	
	album_Id int not null references album(id)
)

create table collection (
	id SERIAL primary key,		
	name varchar(120) not null,
	year_release date not null
)


create table collectionTrack (
	collection_id int not null references collection(id),
	track_id int not null references track(id),
	primary key(collection_id, track_id)
)

create table departaments (
